1.创建存储过程
-- 创建前先删除，养成好习惯
drop PROCEDURE if exists  test;
create PROCEDURE test()
begin
	-- 声明变量 名字i 类型是int 默认值是1
	DECLARE i int DEFAULT 1;
	-- 在mysql里，对于变量的赋值 要加set
	set i=10;
	select i;
	-- 循环用法1
	while i>1 do
		set i=i-1;
		select i;
	end while;
end;

-- 循环用法2
drop PROCEDURE if exists  test2;
create procedure test2()
begin 
	DECLARE i int DEFAULT 1;
	lo:loop
		set i=i+1;
		select i;
		if i>10 then
			LEAVE lo;
		end if;
	end loop lo;
end;
call test2();

	LEAVE lo 
		相当于java的break;
	ITERATE lo
		相当于java的continue;
		
-- 循环用法3
drop PROCEDURE if exists  test3;
create procedure test3()
begin 
	DECLARE i int DEFAULT 1;
	REPEAT
		set i=i+1;
		until i>10
	END REPEAT;
	select i;
end;
call test3();

2.参数
	in    输入参数，无法输出
	out	  输出参数，无法输入
	inout 即输出，也可输入
drop procedure  if exists test4;
create PROCEDURE test4(in a int,out b int)
begin
  -- 给b赋值，因为b属于out
	set b=a+10;
	-- 给a赋值，不会影响到外部的a，因为a属于in
	set a=100;
	-- 无法输出b，因为b是out
	select b;
end;

set @b = 10;
set @a =10;
call test4(@a,@b);
select @b;

	create PROCEDURE printScore(in iname varchar(255),out oscore int)
	begin 
		select score into oscore from grade a,student s where 
		a.student_id=s.id and s.name =iname limit 1;
	end;

	set @score=0;
	call printScore('亚索',@score);
	select @score;


4.游标
create PROCEDURE pro_youbiao()
BEGIN
  declare sno char(50);
	DECLARE sname char(50);
	declare f BOOLEAN DEFAULT true;
  DECLARE youbiao cursor for select st.s,st.sname from student st;
	/*终止while循环*/
  DECLARE CONTINUE HANDLER for not found set f=false;  
  open youbiao;
  while f do
			fetch youbiao into sno,sname;
			if sno='s002' then
       select sno as 'rrrrr';
			else
         select sno as 'zzzz';
      end if;
  end while;
  CLOSE youbiao;
END;
